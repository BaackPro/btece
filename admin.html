Les données affichées sont simulées. Pour des données réelles, intégrez le code de tracking sur votre site.

Intégration du code de tracking

Pour commencer à collecter des données, ajoutez le code suivant juste avant la balise </body> de chaque page de votre site web :

// Code de tracking BTECE Analytics
(function() {
  var btece = window.btece = window.btece || [];
  if (btece.initialize) return;

  // Configuration
  btece.methods = ['page', 'track', 'identify', 'reset'];
  btece.factory = function(method) {
    return function() {
      var args = Array.prototype.slice.call(arguments);
      args.unshift(method);
      btece.push(args);
      return btece;
    };
  };

  // Création des méthodes
  for (var i = 0; i < btece.methods.length; i++) {
    var key = btece.methods[i];
    btece[key] = btece.factory(key);
  }

  // Track page view
  btece.page();

  // Envoi des données au serveur
  setInterval(function() {
    if (btece.length > 0) {
      // Simuler l'envoi des données
      console.log('Envoi des données analytics:', btece);
      btece.splice(0, btece.length);
    }
  }, 10000);

  btece.initialize = true;
})();

Ce code collectera automatiquement les informations suivantes :

  • Pages visitées
  • Durée des visites
  • Type d'appareil et navigateur
  • Localisation géographique approximative
  • Référents (sites source)

Configuration des variables d'environnement

Pour configurer les identifiants d'administration, créez un fichier .env à la racine de votre projet avec le contenu suivant :

# Fichier de configuration .env
# Identifiants d'administration
ADMIN_USERNAME="votre_nom_utilisateur"
ADMIN_PASSWORD="votre_mot_de_passe_complexe"
ADMIN_EMAIL="admin@btece.com"

# Paramètres de l'application
APP_ENV="production"
APP_DEBUG="false"
APP_URL="https://btece.netlify.app"

# Clé secrète pour les sessions (optionnel)
SESSION_SECRET="votre_cle_secrete_tres_longue_ici"

Important : Le fichier .env ne doit jamais être partagé ou commit dans votre dépôt Git. Ajoutez-le à votre .gitignore.

Configuration sur Netlify

Si vous hébergez sur Netlify, vous pouvez configurer les variables d'environnement dans l'interface d'administration :

  1. Allez dans Site settings > Environment variables
  2. Ajoutez les variables ADMIN_USERNAME et ADMIN_PASSWORD avec vos valeurs
  3. Déployez à nouveau votre site
